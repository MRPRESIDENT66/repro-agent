    --master_addr=$MASTER_ADDR \
    --nproc_per_node=$GPUS \
    --master_port=$PORT \

#!/usr/bin/env bash

JOB_NAME=$2

CHECKPOINT=$4
GPUS=${GPUS:-8}

CPUS_PER_TASK=${CPUS_PER_TASK:-5}
PY_ARGS=${@:5}
